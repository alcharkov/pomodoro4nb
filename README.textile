p<. +*Intallation*+

p<. _From *.NBM file_

# "Download latest version":http://plugins.netbeans.org/plugin/40582/pomodoro4nb
# Open Plugin Manager (Select _Tools->Plugin_)
# Open "_Downloaded_" tab and select downloaded *.nbm file  
# Click "_Intall_" button

p<. _From Update Center_

# Open Plugin Manager (Select _Tools->Plugin_)
# Select "_Available Plugins_" tab and enter "pomodoro4nb" to search field
# Check corresponding plugin in available list and click "_Intall_" button

p<. _From Source Code_

# Chekout sources
# Open NetBeans project
# Right click on project and  select "_Install/Reload in Development IDE_" in context menu

p<. +*Usages*+

# Add one or more tasks to the task list
# Choose a task and start the timer
# Work on the task until the pomodoro rings, then check your task list
# Take a short break
# Every 4 pomodoros take a longer break.
